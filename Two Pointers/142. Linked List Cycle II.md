Given a linked list, return the node where the cycle begins. If there is no cycle, return null.

My answer

```ruby
# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def detectCycle(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        nodeseen = set([])
        while head != None:
            if head in nodeseen:
                return head
            else:
                nodeseen.add(head)
            head = head.next
        return None
```
